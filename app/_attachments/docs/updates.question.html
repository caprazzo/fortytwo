1 /**
  2 	Handler for a newly submitted question
  3 	@name question
  4 	@class
  5 	@memberOf updates
  6 */
  7 function(doc, req) {
  8 	// !code _attachments/lib/fortytwo.js
  9 	// !code vendor/couchapp/date.js
 10 	// update
 11 	if (doc) {
 12 		doc.updated = new Date().toJSON();
 13 		return [doc, '{"ok":"updated"}'];
 14 	}
 15 	// new question
 16 	else {
 17 		// trunk uses req.id, 0.10 req.docId
 18 		if (!req.id) req.id = req.docId;
 19 		var now = new Date();
 20 		doc = eval('('+req.body+')');
 21 		doc.created = now.toJSON();
 22 		doc._id = req.id;
 23 		if (doc.tags) {
 24 			var tags = []
 25 			doc.tags.split(',').forEach(function(tag) {
 26 				tags.push(tag.replace(/^\s*/,'').replace(/\s*$/,''));
 27 			});
 28 			doc.tags = tags;
 29 		}
 30 		return [doc, '{"ok":true, "id":"'+doc._id+'"}'];
 31 	}
 32 }
