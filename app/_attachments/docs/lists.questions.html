1 /**
  2 	Used with {@link views.questions}?<strong>descending=true</strong>
  3 	will show most recent questions first and a count of answers for each question.
  4 	@name lists.questions
  5 */
  6 function(head, req) {
  7 // !json templates
  8 // !code lib/mustache.js
  9 
 10 	start({"headers":{"Content-Type" : "text/html; charset=utf-8"}});
 11 
 12 	var preview_length=100;	
 13 	send(Mustache.to_html(templates.head, {title:"questions"}));
 14 	send(Mustache.to_html(templates.questions.list_head));
 15 	answers = 0;
 16 	while (row = getRow()) {
 17 		if (row.value.answer) {
 18 			answers++;
 19 		}	
 20 		else if (row.value) {
 21 			var question = row.value;
 22 			question.answers = answers;
 23 			
 24 			question.question_preview = (question.question.length > preview_length) 
 25 				? question.question.substring(0,100) + '...'
 26 				: question.question;
 27 			
 28 			question.url_id = encodeURIComponent(question._id).replace(/%22/g,"%5C%22");	
 29 			
 30 			send(Mustache.to_html(templates.questions.list_item, question));
 31 			answers = 0;
 32 		} 	
 33 	}
 34 	
 35 	send(Mustache.to_html(templates.questions.list_foot));
 36 	send(Mustache.to_html(templates.foot),{});
 37 		
 38 }
