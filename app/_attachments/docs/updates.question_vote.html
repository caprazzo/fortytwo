1 /**
  2 	<p>Handler for adding or subtracting votes from questions.</p>
  3 	
  4 	<p>A query parameter vote=[up|down] must be supplied.</p>
  5 	
  6 	@name updates.question_vote
  7 */
  8 function(doc, req) {
  9 	if (!doc || !req.query || !req.query.vote) {
 10 		return [null, '{"error":"invalid request"}'];
 11 	}
 12 	
 13 	if (typeof doc.votes != 'number') doc.votes = 1;	
 14 	if (req.query.vote == 'up') {
 15 		doc.votes++;
 16 	}
 17 	else if (req.query.vote == 'down') {
 18 		doc.votes--;	
 19 	}
 20 	return [doc, toJSON({
 21 		ok: "voted " + req.query.vote,
 22 		votes: doc.votes
 23 	})];
 24 	
 25 }
