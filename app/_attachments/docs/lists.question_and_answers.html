  3 	will show a question and all its answers.
  4 	@name lists.question_and_answers
  5 */
  6 function(head, req) {
  7 
  8 // !json templates
  9 // !code lib/mustache.js	
 10 // !code lib/showdown.js
 11 // !code _attachments/lib/fortytwo.js
 12 
 13 	start({"headers":{"Content-Type" : "text/html; charset=utf-8"}});
 14 	var converter = new Showdown.converter();

 25 		}

 29 		}
 30 	}	
 31 	send(Mustache.to_html(templates.answers.list_foot));
 32 	send(Mustache.to_html(templates.answer_form, question_doc));
 33 	send(Mustache.to_html(templates.app_foot));

 35 	Used with {@link views.questions}?<strong>descending=true</strong>
 36 	will show most recent questions first and a count of answers for each question.
 37 	@name lists.questions
 38 */
 39 function(head, req) {
 40 // !json templates
 41 // !code lib/mustache.js
 42 // !code lib/showdown.js
 43 // !
 44 	start({"headers":{"Content-Type" : "text/html; charset=utf-8"}});
 45 
 46 	var preview_length=100;	
 47 	send(Mustache.to_html(templates.app_head, {title:"questions"}));
 48 	send(Mustache.to_html(templates.questions.list_head));
 49 	answers = 0;
 50 	var converter = new Showdown.converter();
 51 
 52 	while (row = getRow()) {
 53 		if (row.value==1) {
 54 			answers++;
 55 		}	
 56 		else if (row.value) {
 57 			var question = row.value;
 58 			question.answers = answers;
 59 			
 60 			question.question_preview = (question.question.length > preview_length) 
 61 				? converter.makeHtml(question.question.substring(0,100)) + '...'
 62 				: converter.makeHtml(question.question);	
 63 			
 64 			question.url_id = encodeURIComponent(question._id).replace(/%22/g,"%5C%22");	
 65 			
 66 			send(Mustache.to_html(templates.questions.list_item, question));
 67 			answers = 0;
 68 		} 	
 69 	}
 70 	
 71 	send(Mustache.to_html(templates.questions.list_foot));
 72 	send(Mustache.to_html(templates.app_foot),{});
 73 		
 74 }

 76 /**
 77 	<p>If invoked with include_docs=true, this view returns a full answer doc and the full question for each
 78 	answer. It's another way to see linked documents together in the client.</p>
 79 	
 80 	<p>I used this view, along with some python to copy question.created into answer.question_created for
 81 	each (question, answer) where question._id == answer.question_id
 82 	
 83 	python code is in resync.py in the root of the project
 84 	
 85 	@name views.all_answers.map
 86 */
 87 function(doc) {
 88 	if(doc.answer) {
 89 		emit(null, {_id:doc.question_id, answer_doc:doc});
 90 	}
 91 }/**
 92 	<p>Emits <strong>[question.id, 0]</strong> when the document is a question
 93 	and <strong>[answer.question_id,answer.created]</strong> when it's an answer.</p>
 94 	
 95 	<p>
 96 		The first key level keeps questions and answers adjacent, the second keeps them in date order.
 97 		As questions are always created before answers, this also makes questions always appear before answers.
 98 	</p>
 99 	
100 	<p>
101 		If queried with startkey=["<question_id>"]&endkey=["<question_id>","Z"] it will return one question
102 		followed by all its answers. The "Z" in the endkey is always higher 
103 		than a date as stored in this application
104 	</p>	
105 	
106 	<p>A sample key sequence is</p>
107 	<ul>
108 		<li>[question-y, <strong>'2010/01/01 12:12:37 +0000'] // question</strong></li>
109 		<li>[question-y, '2010/01/03 12:12:37 +0000'] // answer</li>
110 		<li>[question-y, '2010/01/03 12:12:38 +0000'] // answer</li>
111 		<li>[question-x, <strong>'2010/12/01 12:12:37 +0000'] // question</strong></li>
112 		<li>[question-x, '2010/12/04 12:12:37 +0000'] // answer</li>
113 		<li>[question-x, '2010/12/04 12:12:37 +0000'] // answer</li>
114 	</ul>
115 	
116 	
117 
118 	<p>For more about joins, see
119 	see http://www.cmlenz.net/archives/2007/10/couchdb-joins</p>
120 	
121 	@name views.questions.map
122 */
123 function(doc) {
124 	if (doc.question) {
125 		emit([doc._id, doc.created], doc);
126 	}
127 	if (doc.answer) {
128 		emit([doc.question_id, doc.created], doc);
129 	}
130 }/**
131 	Emits <strong>[question.created, question.id, 0]</strong> when the document is a question
132 	and <strong>[answer.question_created, answer.question.id,1]</strong> when it's an answer.
133 		
134 	answer.question_created is always equal to question.created, so a sample key sequence is:
135 	<ul>
136 		<li>[2009-01-01, question-y, <strong>0] // question</strong></li>
137 		<li>[2009-01-01, question-y, 1] // answer</li>
138 		<li>[2009-01-01, question-y, 1] // answer</li>		
139 		<li>[2010-01-02, question-x, <strong>0] // question</strong></li>
140 		<li>[2010-01-02, question-x, 1] // answer</li>
141 		<li>[2010-01-02, question-x, 1] // answer</li>
142 	</ul>
143 
144 	The second numeric key forces the answers to follow the question,
145 	see http://www.cmlenz.net/archives/2007/10/couchdb-joins for more about "joins".
146 	
147 	@name views.questions_index.map
148 */
149 function(doc) {
150 	if (doc.question) {
151 		emit([doc.created, doc._id, 0], doc);
152 	}
153 	else if (doc.answer) {
154 		emit([doc.question_created, doc.question_id, 1], 1);
155 	}

185 	Handler for a newly submitted answer
186 	<strong>must be invoked for an existing question</strong>	
187 	@name answer
188 	@class
189 	@memberOf updates
190 */
191 function(doc, req) {
192 	// !code _attachments/lib/fortytwo.js
193 	// !code vendor/couchapp/date.js	
194 	// the existing doc must be a question
195 	if (doc) {
196 		var answer = eval('('+req.body+')');
197 		
198 		// store servertime...
199 		answer.created = (new Date()).toJSON();
200 		
201 		// copy question creation date
202 		answer.question_created = doc.created;
203 		
204 		answer.question_id = doc._id;
205 		
206 		var reply = {
207 			ok: 'answer accepted',
208 			answer: answer
209 		}
210 		
211 		return [answer, toJSON(reply)];
212 	}
213 	throw('question must exist');
214 }/**
215 	Handler for a newly submitted question
216 	@name question
217 	@class
218 	@memberOf updates
219 */
220 function(doc, req) {
221 	// !code _attachments/lib/fortytwo.js
222 	// !code vendor/couchapp/date.js
223 	// update
224 	if (doc) {
225 		doc.updated = new Date().toJSON();
226 		return [doc, '{"ok":"updated"}'];
227 	}
228 	// new question
229 	else {
230 		// trunk uses req.id, 0.10 req.docId
231 		if (!req.id) req.id = req.docId;
232 		var now = new Date();
233 		doc = eval('('+req.body+')');
234 		doc.created = now.toJSON();
235 		doc._id = req.id;
236 		if (doc.tags) {
237 			var tags = []
238 			doc.tags.split(',').forEach(function(tag) {
239 				tags.push(tag.replace(/^\s*/,'').replace(/\s*$/,''));
240 			});
241 			doc.tags = tags;
242 		}
243 		return [doc, '{"ok":true, "id":"'+doc._id+'"}'];
244 	}
